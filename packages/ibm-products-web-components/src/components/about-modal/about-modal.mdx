export const customFunctionArr= [
      Object.entries(helpers)
    .map(([name, value]) => {
      // If it's a function -> stringify function normally
      if (typeof value === 'function') {
        return `const ${name} = ${value.toString()}`;
      }
      // Otherwise treat as literal â†’ add quotes properly
      return `const ${name} = ${JSON.stringify(value)};`;
    })
    .join('\n')
];
